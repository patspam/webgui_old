Class YAHOO.ext.SplitBar.AbsoluteLayoutAdapter

Package:YAHOO.ext
Class:SplitBar.AbsoluteLayoutAdapter
Extends:SplitBar.BasicLayoutAdapter
Defined In:SplitBar.js
